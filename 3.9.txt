p = int(input("Введите количество секунд, прошедших с начала суток: "))
hours = p // 3600
minutes = (p % 3600) // 60
seconds = (p % 3600) % 60
print("С начала суток прошло", hours, "полных часов")
print("С начала очередного часа прошло", minutes, "полных минут")
print("С начала очередной минуты прошло", seconds, "полных секунд")
