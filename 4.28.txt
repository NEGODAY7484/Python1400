number = int(input("Введите двузначное число: "))

# вычисляем сумму цифр числа
sum_of_digits = number // 10 + number % 10

# проверяем, является ли сумма цифр двузначным числом
if sum_of_digits >= 10 and sum_of_digits <= 99:
    print("Сумма цифр является двузначным числом")
else:
    print("Сумма цифр не является двузначным числом")

# проверяем, больше ли число а сумма его цифр
if number > sum_of_digits:
    print("Число больше суммы его цифр")
else:
    print("Число не больше суммы его цифр")
