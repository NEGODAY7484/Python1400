a = int(input("Введите размер a стола: "))
b = int(input("Введите размер b стола: "))
c = int(input("Введите размер c картона: "))
d = int(input("Введите размер d картона: "))

# Вычисляем количество прямоугольников, размещенных длинной стороной вдоль длинной стороны стола
n1 = (a // c) * (b // d)

# Вычисляем количество прямоугольников, размещенных длинной стороной вдоль короткой стороны стола
n2 = (a // d) * (b // c)

# Выводим результаты
if n1 > n2:
    print("На столе можно разместить большее количество картонных прямоугольников, размещенных длинной стороной вдоль длинной стороны стола.")
elif n1 < n2:
    print("На столе можно разместить большее количество картонных прямоугольников, размещенных длинной стороной вдоль короткой стороны стола.")
else:
    print("На столе можно разместить одинаковое количество картонных прямоугольников, независимо от ориентации.")
