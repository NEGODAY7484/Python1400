a = float(input("Введите длину стороны a: "))
b = float(input("Введите длину стороны b: "))
c = float(input("Введите длину стороны c: "))
d = float(input("Введите длину стороны d: "))

if (a <= c and b <= d) or (a <= d and b <= c):
    print("Прямоугольник со сторонами {}, {} может быть помещён в прямоугольник со сторонами {}, {}.".format(a, b, c, d))
else:
    print("Прямоугольник со сторонами {}, {} не может быть помещён в прямоугольник со сторонами {}, {}.".format(a, b, c, d))
