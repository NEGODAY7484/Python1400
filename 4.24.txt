number = int(input("Введите натуральное число: "))  # преобразуем введенную пользователем строку в число

if number % 2 == 0:  # проверяем, является ли число четным
    print("Число {} является четным".format(number))
else:
    print("Число {} не является четным".format(number))

if number % 10 == 7:  # проверяем, оканчивается ли число на 7
    print("Число {} оканчивается на 7".format(number))
else:
    print("Число {} не оканчивается на 7".format(number))
