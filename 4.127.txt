a = int(input("Введите первое целое число: "))
b = int(input("Введите второе целое число: "))
c = int(input("Введите третье целое число: "))

# Находим наибольшее число
if a > b and a > c:
    print(a, "является наибольшим числом")
elif b > a and b > c:
    print(b, "является наибольшим числом")
else:
    print(c, "является наибольшим числом")

# Находим наименьшее число
if a < b and a < c:
    print(a, "является наименьшим числом")
elif b < a and b < c:
    print(b, "является наименьшим числом")
else:
    print(c, "является наименьшим числом")

# Находим среднее число
if (a < b and a > c) or (a > b and a < c):
    print(a, "является средним числом")
elif (b < a and b > c) or (b > a and b < c):
    print(b, "является средним числом")
else:
    print(c, "является средним числом")
