a = float(input("Введите первую сторону треугольника: "))
b = float(input("Введите вторую сторону треугольника: "))
c = float(input("Введите третью сторону треугольника: "))

if a == b == c:
    print("Треугольник является равносторонним")
elif a == b or b == c or a == c:
    print("Треугольник является равнобедренным")
else:
    print("Треугольник не является ни равносторонним, ни равнобедренным")
