# вводим данные
R1 = float(input("Введите сопротивление R1: "))
U1 = float(input("Введите напряжение U1: "))
R2 = float(input("Введите сопротивление R2: "))
U2 = float(input("Введите напряжение U2: "))

# вычисляем ток на каждом участке цепи
I1 = U1 / R1
I2 = U2 / R2

# выводим результат
if I1 < I2:
    print("На участке R1 протекает меньший ток:", I1)
else:
    print("На участке R2 протекает меньший ток:", I2
