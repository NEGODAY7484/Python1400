t = float(input("Введите время t в минутах, прошедшее с начала очередного часа: "))

# Вычисляем остаток от деления времени на длительность цикла светофора (8 минут)
remainder = t % 8

# Определяем цвет сигнала светофора для пешеходов в данный момент и выводим результат
if remainder < 3:
    print("Горит зеленый сигнал для пешеходов.")
elif remainder < 5:
    print("Горит красный сигнал для пешеходов.")
else:
    print("Горит зеленый сигнал для пешеходов.")
