p = int(input("Введите натуральное число, большее 9: "))

if p <= 9:
    print("Число должно быть больше 9.")
else:
    # находим количество единиц
    ones = p % 10

    # находим количество десятков
    tens = (p % 100) // 10

    print("Количество единиц:", ones)
    print("Количество десятков:", tens)
