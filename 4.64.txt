a = float(input("Введите ширину форточки (в см): "))
b = float(input("Введите высоту форточки (в см): "))
d = float(input("Введите диаметр головы (в см): "))

# Добавляем зазоры
a -= 2
b -= 2

# Проверяем, может ли голова пройти через форточку
if d <= min(a, b):
    print("Вася сможет высунуть голову в форточку")
else:
    print("Вася не сможет высунуть голову в форточку")
