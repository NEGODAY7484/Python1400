num = input('Введите натуральное число: ')
if num[-1] in ['0', '2', '4', '6', '8']:
    print('Число заканчивается четной цифрой')
else:
    print('Число заканчивается нечетной цифрой')
if num[-1] in ['1', '3', '5', '7', '9']:
    print('Число заканчивается нечетной цифрой')
else:
    print('Число заканчивается четной цифрой')
