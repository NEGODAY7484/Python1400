a = float(input("Введите первое вещественное число: "))
b = float(input("Введите второе вещественное число: "))
c = float(input("Введите третье вещественное число: "))

# Находим наибольшее число
if a > b and a > c:
    print(a, "является наибольшим числом")
elif b > a and b > c:
    print(b, "является наибольшим числом")
else:
    print(c, "является наибольшим числом")

# Находим наименьшее число
if a < b and a < c:
    print(a, "является наименьшим числом")
elif b < a and b < c:
    print(b, "является наименьшим числом")
else:
    print(c, "является наименьшим числом")
