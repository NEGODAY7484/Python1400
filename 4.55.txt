num = int(input("Введите четырехзначное число: "))

# а) проверка на наличие цифр 2 или 7
if '2' in str(num) or '7' in str(num):
    print("Цифры 2 или 7 входят в число")
else:
    print("Цифры 2 или 7 не входят в число")

# б) проверка на наличие цифр 3, 6 или 9
if '3' in str(num) or '6' in str(num) or '9' in str(num):
    print("Цифры 3, 6 или 9 входят в число")
else:
    print("Цифры 3, 6 или 9 не входят в число")
