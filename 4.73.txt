a2 = int(input("Введите число десятков: "))
a1 = int(input("Введите число единиц: "))
b = int(input("Введите однозначное число: "))

diff = (a2*10 + a1) - b # находим разность

result = [diff//10, diff%10] # разбиваем разность на цифры

print("Цифры числа, равного разности заданных чисел: ", result[0], result[1])
