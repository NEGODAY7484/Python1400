number = int(input("Введите двузначное число: "))
digit1 = number // 10 # первая цифра
digit2 = number % 10 # вторая цифра
square = number ** 2 # квадрат числа
sum_of_cubes = 4 * (digit1 ** 3 + digit2 ** 3) # учетверенная сумма кубов цифр

if square == sum_of_cubes:
    print("Квадрат числа равен учетверенной сумме кубов его цифр")
else:
    print("Квадрат числа не равен учетверенной сумме кубов его цифр")
