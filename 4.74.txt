a1 = int(input("Введите число единиц первого двузначного числа: "))
a2 = int(input("Введите число десятков первого двузначного числа: "))
b1 = int(input("Введите число единиц второго двузначного числа: "))
b2 = int(input("Введите число десятков второго двузначного числа: "))

raznost = abs((a2 * 10 + a1) - (b2 * 10 + b1))

print("Разность заданных чисел:", raznost)
print("Цифра десятков в полученном числе:", raznost // 10)
print("Цифра единиц в полученном числе:", raznost % 10)
