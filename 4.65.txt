a = float(input("Введите длину ребра а: "))
b = float(input("Введите длину ребра b: "))
c = float(input("Введите длину ребра c: "))
x = float(input("Введите длину стороны x прямоугольного отверстия: "))
y = float(input("Введите длину стороны y прямоугольного отверстия: "))

if (a+2 <= x and max(b,c) <= y-2) or (b+2 <= x and max(a,c) <= y-2) or (c+2 <= x and max(a,b) <= y-2):
  print("Кирпич проходит в прямоугольное отверстие")
else:
  print("Кирпич не проходит в прямоугольное отверстие")
