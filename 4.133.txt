day_num = int(input("Введите порядковый номер дня недели (1-7): "))

if day_num == 1:
    print("Понедельник")
elif day_num == 2:
    print("Вторник")
elif day_num == 3:
    print("Среда")
elif day_num == 4:
    print("Четверг")
elif day_num == 5:
    print("Пятница")
elif day_num == 6:
    print("Суббота")
elif day_num == 7:
    print("Воскресенье")
else:
    print("Ошибка: введите число от 1 до 7")
